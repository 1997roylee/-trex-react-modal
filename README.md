## Install

```bash
npm install --save @trex/react-modal
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from '@trex/react-modal'
import '@trex/react-modal/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
